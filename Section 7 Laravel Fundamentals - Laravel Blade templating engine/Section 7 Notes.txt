- add an array variable in contact method in PostsController in App/Http/Controllers/PostsController.php
* content of these array will be displayed using @foreach

public function contact() {
    $people = ['Edwin', 'Jose', 'James', 'Peter', 'Maria'];
    return view('contact', compact('people'));
}
